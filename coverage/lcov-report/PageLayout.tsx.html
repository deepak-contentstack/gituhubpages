const LayoutHeader = (props: layoutHeaderProps) => {
  return useMemo(() => {
    if (props.header !== undefined) {
      return (
        <div className={`PageLayout__head ${(props.type === 'card') ? (props.headerType === 'custom') ? 'PageLayout__head--card--custom-header' : 'PageLayout__head--card' : ''}`}>
          {props.toggleStat !== undefined ? (
            <div
              className="PageLayout__head__hamburger"
              onClick={() => props.toggle()}
            ><Icon icon="Hamburger" size="original" /></div>
          ) : (
            <div className="PageLayout__head__backArrow" onClick={(event: any) => props.header.backNavigation(event)}>
              <Icon icon="BackArrow" size="original" />
            </div>
          )}
          {/* {props.header.backText !== undefined ? <div onClick={(event: any) => props.header.backNavigation(event)} className="PageLayout__head__backText">{props.header.backText}</div> : ''} */}
          {props.header.component !== undefined ? props.header.component : ''}
          {/* {props.header.Breadcrumb !== undefined ? <div className="breadcrumb">{props.header.Breadcrumb} </div> : null} */}
        </div>
      )
    } else {
      return null
    };
 
  }, [props.header && props.header.title, props.toggleStat, props.header && props.header.component]);
 
};

const Content = (props: contentProps) => {
  return useMemo(() => {
    if (props.content !== undefined) {
      return (
        <>
          {props.type === 'edit' ? (
            <div className={`PageLayout__body`} id="PageLayout__body">
              {props.content.livePreview}
              <div className="PageLayout__body__container">
                {props.content.component}
              </div>
            </div>
          ) :
            <div className={`PageLayout__body ${(props.type === 'card') ? (props.headerType === 'custom') ? 'PageLayout__body--card--custom-header' : 'PageLayout__body--card' : ''}`}>
              {props.content.component}
            </div>
          }
 
 
        </>
      )
    } else {
      return null
    };
  }, [props.content, props.toggleStat]);
};
 
const ListLayout = (props: primaryLayoutProps) => {
  const [toggleLeftNav, setLeftNav] = useState(true);
  const sidebar = useRef(null);
  const [toggleLayout, setToggleLayout] = useState(false);
  const ele: any = document.getElementsByClassName('PageLayout__content');
 
  const handleToggle = () => {
    // setLeftNav(!toggleLeftNav)
    // if (toggleLeftNav === true) {
    //   ele.classList.add('collapse-out')
    // } else {
    //   ele.classList.remove('collapse-out')
    // }
    sidebar.current.classList.toggle('PageLayout__leftSidebar--hide');
    sidebar.current.classList.toggle('PageLayout__leftSidebar--show');
    let leftSidebarElem: any = document.getElementsByClassName('PageLayout__leftSidebar')[0];
    if (sidebar.current.classList.contains('PageLayout__leftSidebar--show')) {
      console.log("pagelayout 1");
      setTimeout(function () {
        leftSidebarElem.style.zIndex = "8";
      }, 300);
    } else {
      console.log("pagelayout 2");
      leftSidebarElem.style.zIndex = "2";
    }
    setToggleLayout(!toggleLayout);
  };
 
 
  return (
    <div className={`PageLayout PageLayout--primary ${props.className ? props.className : ''}`}>
      {props.leftSidebar !== undefined ? (
        <div ref={sidebar} className={`PageLayout__leftSidebar ${toggleLeftNav === true ? 'PageLayout__leftSidebar--show' : 'PageLayout__leftSidebar--hide'}`} >
          <LeftSidebar leftSidebar={props.leftSidebar} />
        </div>
      ) : null}
      <div
        className={`PageLayout__content ${props.leftSidebar === undefined ? 'PageLayout__content--full' : ''} ${(props.type === 'card') ? (props.headerType === 'custom') ? 'PageLayout__content--card--custom-header' : 'PageLayout__content--card' : ''} ${toggleLayout ? 'PageLayout__content--full' : ''}`}
      >
        {props.leftSidebar !== undefined ? (
          <LayoutHeader
            header={props.header}
            type={props.type}
            headerType={props.headerType}
            toggleStat={toggleLeftNav}
            toggle={handleToggle}
          />
        ) : (
          <LayoutHeader header={props.header} headerType={props.headerType} toggle={handleToggle} type={props.type} />
        )}
        <Content content={props.content} type={props.type} headerType={props.headerType} toggle={handleToggle} toggleStat={toggleLeftNav} />
      </div>
    </div>
  );
};

const FullLayout = (props: secondaryLayoutProps) => {
  const { className, header, mode, leftSidebar, rightSidebar, content, footer, layoutBackgroundColor, rightFooter, leftFooter, topNavBar } = props
  return (
    <div className={`PageLayout PageLayout--secondary ${mode == 'fullscreen' ? 'PageLayout--full-screen' : ''} ${!leftFooter ? `${footer ? '' : 'PageLayout--content-fullpage'}` : ''} ${layoutBackgroundColor === true ? 'PageLayout--secondary-edit' : ''} ${className ? className : ''} `}>
      <LayoutHeader header={header} type={'edit'} mode={mode} />
      <div className={`PageLayout__content`}>
        {leftSidebar && (
          <div className={`PageLayout__leftSidebar`}>
            <LeftSidebar leftSidebar={leftSidebar} />
          </div>
        )}
        {rightSidebar && (
          <div className={`PageLayout__rightSidebar`}>
            <RightSidebar rightSidebar={rightSidebar} />
          </div>
        )}
        {topNavBar && (
          <div className={`PageLayout__topNavbar`}>
            <TopNavbar topNavBar={topNavBar} />
          </div>
        )}
        <Content content={content} type={'edit'} mode={mode} />
      </div>
      {footer && (
        <div className={`PageLayout__footer`}>
          <Footer footer={footer} />
        </div>
      )}
      {rightFooter && (
        <RightFooter rightFooter={rightFooter} />
      )}
      {leftFooter && (
        <LeftFooter leftFooter={leftFooter} />
      )}
    </div>
  );
};
 
const EmptyPage = (props: emptyPageProps) => {
  const PageHeaderWithCSLogo = () => (
    <div className='PageLayout__head--emptyLayout'>
      <div
        className={`PageLayout__head--CSLogo ${props.header?.headerLogoOnClick ? 'clickable' : ''}`}
        onClick={props.header?.headerLogoOnClick}
      >
        <Icon icon='CSLogo' />
      </div>
    </div>
  )
 
  return (
    <div
      className={
        `PageLayout--emptyLayout
        ${props.content.backgroundColor === 'secondary' ? 'secondaryBackground' : 'primaryBackground'}`
      }>
      <PageHeaderWithCSLogo />
      <div className={
        `${props.content.alignContent === 'center' ? 'PageLayout__content--center' : 'PageLayout__content'}`
      }>
        {props.content.component}
      </div>
    </div>
  )
}
 
const PageLayout = (props: layoutProps) => {
  const { header, leftSidebar, content, type, rightSidebar, footer, headerType, className, mode, layoutBackgroundColor, rightFooter, leftFooter, topNavBar } = props;
 
  switch (type) {
    case 'list':
      return <ListLayout header={header} leftSidebar={leftSidebar} content={content} className={className} />
    case 'card':
      return <ListLayout header={header} headerType={headerType} leftSidebar={leftSidebar} content={content} type={type} className={className} />
    case 'edit':
      return (
        <FullLayout
          header={header}
          leftSidebar={leftSidebar}
          content={content}
          rightSidebar={rightSidebar}
          footer={footer}
          className={className}
          mode={mode}
          layoutBackgroundColor={layoutBackgroundColor}
          rightFooter={rightFooter}
          leftFooter={leftFooter}
          topNavBar={topNavBar}
        />
      )
    case 'empty':
      return <EmptyPage content={content} header={header} />
    default:
      return null
  };
};
 
export default PageLayout;
